function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = { weekly: getSheetData(ss, 'Weekly_Data'), raci: getSheetData(ss, 'RACI_Status'), gantt: getSheetData(ss, 'Gantt_Status') }; const json = JSON.stringify(data); const callback = e && e.parameter && e.parameter.callback; if (callback) { return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const body = JSON.parse(e.postData.contents || '{}'); const type = body.type; const payload = body.payload || {}; if (type === 'weekly') { upsertRow(ss, 'Weekly_Data', 'Week', payload); } if (type === 'raci') { upsertRow(ss, 'RACI_Status', 'Work_Item', payload); } if (type === 'gantt') { upsertRow(ss, 'Gantt_Status', 'Task_ID', payload); } return ContentService .createTextOutput(JSON.stringify({ success: true, type: type })) .setMimeType(ContentService.MimeType.JSON); } function getSheetData(ss, sheetName) { const sheet = ss.getSheetByName(sheetName); if (!sheet) return []; const values = sheet.getDataRange().getValues(); if (values.length < 2) return []; const headers = values.shift(); return values .filter(row => row.some(cell => cell !== '')) .map(row => { const obj = {}; headers.forEach((header, index) => { obj[header] = row[index]; }); return obj; }); } function upsertRow(ss, sheetName, keyColumn, payload) { const sheet = ss.getSheetByName(sheetName); if (!sheet) throw new Error('Missing sheet: ' + sheetName); const values = sheet.getDataRange().getValues(); const headers = values[0]; const keyIndex = headers.indexOf(keyColumn); if (keyIndex === -1) throw new Error('Missing key column: ' + keyColumn); const keyValue = payload[keyColumn]; if (!keyValue) throw new Error('Missing key value for: ' + keyColumn); let rowIndex = -1; for (let i = 1; i < values.length; i++) { if (String(values[i][keyIndex]) === String(keyValue)) { rowIndex = i + 1; break; } } const row = headers.map(header => getPayloadValue(header, payload)); if (rowIndex > -1) { sheet.getRange(rowIndex, 1, 1, row.length).setValues([row]); } else { sheet.appendRow(row); } } function getPayloadValue(header, payload) { if (payload[header] !== undefined) return payload[header]; const aliases = { 'Leads (MQL)': 'Leads', 'Sales (SQL)': 'Sales', 'What_Changed': 'What_Changed', 'Work Item': 'Work_Item', 'Current Owner': 'Current_Owner', 'Due Date': 'Due_Date', 'Next Action': 'Next_Action', 'Progress Note': 'Progress_Note' }; const alias = aliases[header]; if (alias && payload[alias] !== undefined) return payload[alias]; return ''; }
